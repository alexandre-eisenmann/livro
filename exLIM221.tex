\begin{align}
&y=\sqrt{x+\sqrt{4x^2+2}}+\sqrt{x}\nonumber\\
%resposta
&y'=\frac{1}{2\sqrt{x+\sqrt{4x^2+2}}}\left(1+\frac{4x}{\sqrt{4x^2+2}}\right)+\frac{1}{2\sqrt{x}}\nonumber\\
&y'=\frac{\sqrt{4x^2+2}=4x}{2\sqrt{(4x^2+2)(x+\sqrt{4x^2+2})}}+\frac{1}{2\sqrt{x}}\nonumber
\end{align}
\end{ex}
