\begin{ex}
\begin{align}
&y=\sqrt{x+\sqrt{1+x^2}}\nonumber\\
%resposta
&y'=\frac{1}{2\sqrt{x+\sqrt{1+x^2}}}.\left(1+\frac{x}{\sqrt{1+x^2}}\right)\nonumber\\
&y'=\frac{1}{2\sqrt{x+\sqrt{1+x^2}}}.\frac{x+\sqrt{1+x^2}}{\sqrt{1+x^2}}=\frac{x+\sqrt{1+x^2}}{2\sqrt{1+x^2}}\nonumber
\end{align}
\end{ex}
