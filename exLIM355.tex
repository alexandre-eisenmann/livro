\begin{ex}
\begin{align}
&\text{Calcular a n'ésima derivada de}\quad y=\ln{(ax+b)}\nonumber\\
%resposta
&y'=\frac{a}{ax+b}\nonumber\\
&y''=\frac{-a^2}{(ax+b)^2}\nonumber\\
&y'''=\frac{2(ax+b)a^2.a}{(ax+b)^4}=\frac{2a^3}{(ax+b)^3}\nonumber\\
&y^{n}=\frac{(-1)^{n-1}(n-1)!a^{n}}{(ax+b)^{n}}\nonumber
\end{align}
\end{ex}
