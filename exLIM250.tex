\begin{ex}
\begin{align}
&y=\sen^2{x}\nonumber\\
%resposta
&y'=2\sen{x}\cos{x}\nonumber\\
&\text{outra maneira de resolver: fazendo}\quad y=\sen{x}\sen{x}\quad\text{e derivando pelo produto, temos:}\nonumber\\
&y'=\cos{x}\sen{x}+\sen{x}\cos{x}=2\sen{x}\cos{x}\nonumber
\end{align}
\end{ex}
