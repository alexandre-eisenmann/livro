\begin{ex}
\begin{align}
&\lim_{x\rightarrow \infty} \frac{x!}{(x+1)!-x!}=\nonumber\\
%resposta
&\lim_{x\rightarrow \infty} \frac{x!}{(x+1)x!-x!}=\lim_{x\rightarrow \infty} \frac{x!}{x![(x+1)-1]}=\nonumber\\
&\lim_{x\rightarrow \infty} \frac{1}{x}=0\nonumber
\end{align}
\end{ex}
