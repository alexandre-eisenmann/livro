&y=\frac{x^2+1}{x-1}\nonumber\\
%resposta
&y'=\frac{(x-1)(x^2+1)'-(x-1)'(x^2+1)}{(x-1)^2}\nonumber\\

&y'=\frac{(x^2-2x-1)}{(x-1)^2}\nonumber
\end{align}
