\begin{ex}
\begin{align}
&y=\frac{1}{x}\nonumber\\
%resposta
&y'=\lim_{h\rightarrow 0} \frac{\frac{1}{x+h}-\frac{1}{x}}{h}=\lim_{h\rightarrow 0} \frac{x-(x+h)}{h(x+h).x}=\nonumber\\
&y'=\lim_{h\rightarrow 0} \frac{-h}{h(x+h).x}=\lim_{h\rightarrow 0} \frac{-1}{(x+h).x}=\frac{-1}{x^2}\nonumber
\end{align}
\end{ex}
