\begin{ex}
\begin{align}
&\lim_{x\rightarrow a+b+c} \frac{(x-a)^2-(b+c)^2}{(x-c)^2-(a+b)^2}=\nonumber\\
%resposta
&\lim_{x\rightarrow a+b+c} \frac{(x-a+b+c)(x-a-b-c)}{(x-c+a+b)(x-a-b-c)}=\nonumber\\
&=\frac{a+b+c-a+b+c}{a+b+c-c+a+b}=\frac{b+c}{a+b}\nonumber
\end{align}
\end{ex}
