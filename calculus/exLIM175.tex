\begin{ex}
\begin{align}
&y=\sqrt{x-1}+\sqrt{x+3}\nonumber\\
%resposta
&y'=(\sqrt{x-1})'+(\sqrt{x+3})'\nonumber\\
&y'=\frac{1}{2}(x-1)^{-\frac{1}{2}}(x-1)'+\frac{1}{2}(x+3)^{-\frac{1}{2}}(x+3)'\nonumber\\
&y'=\frac{1}{2\sqrt{x-1}}+\frac{1}{2\sqrt{x+3}}\nonumber\\
&y'=\frac{\sqrt{x+3}+\sqrt{x-1}}{2\sqrt{x-1}.\sqrt{x+3}}\nonumber
\end{align}
\end{ex}
