\begin{ex}
\begin{align}
&y=x^{\tg{x}}\nonumber\\
%resposta
&\ln{y}=\ln{x}^{\tg{x}}\quad\Rightarrow \ln{y}=\tg{x}.\ln{x}\nonumber\\
&\frac{y'}{y}=\frac{1}{\cos^2{x}}.\ln{x}+\frac{\tg{x}}{x}\nonumber\\
&y'=x^{\tg{x}}\left(\frac{1}{\cos^2{x}}.\ln{x}+\frac{\tg{x}}{x}\right)\nonumber
\end{align}
\end{ex}
