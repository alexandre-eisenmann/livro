\begin{ex}
\begin{align}
&\text{Calcular a n'ésima derivada de} \quad y=(1+x)^m \quad \text{para m>n}\nonumber\\
%resposta
&y'=m(1+x)^{m-1}\nonumber\\
&y''=(m-1)m(1+x)^{m-2}\nonumber\\
&y'''=(m-2)(m-1)m(1+x)^{m-3}\nonumber\\
&y^{n}=(m-1)(m-2).......(m-n+1)(1+x)^{m-n}\nonumber
\end{align}
\end{ex}
