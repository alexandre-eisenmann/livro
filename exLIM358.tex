\begin{ex}
\begin{align}
&\text{Calcular a n'ésima derivada de}\quad y=a^{x}\nonumber\\
%resposta
&y'=a^{x}\log{a}\nonumber\\
&y''=a^{x}(\log{a})(\log{a})=a^{x}\log^2{a}\nonumber\\
&y'''=a^{x}(\log{a})(\log{a})(\log{a})=a^{x}\log^3{a}\nonumber\\
&y^{n}=a^{x}\log^{n}{a}\nonumber
\end{align}
\end{ex}
