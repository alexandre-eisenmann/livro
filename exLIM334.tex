\begin{ex}
\begin{align}
&y=x^{\sen{x}}\nonumber\\
%resposta
&\ln{y}=\ln{x^{\sen{x}}} \Rightarrow \ln{y}=\sen{x}.\ln{x}\nonumber\\
&\frac{y'}{y}=\cos{x}.\ln{x}+\frac{\sen{x}}{x}\nonumber\\
&y'=x^{\sen{x}}.\left(\cos{x}.\ln{x}+\frac{\sen{x}}{x}\right)\nonumber
\end{align}
\end{ex}
