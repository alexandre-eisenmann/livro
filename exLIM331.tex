\begin{ex}
\begin{align}
&y=\sqrt{x}^{x}\nonumber\\
%resposta
&\ln{y}=\ln{\sqrt{x}^{x}}\quad \Rightarrow \ln{y}=x.\ln{\sqrt{x}}\nonumber\\
&\frac{y'}{y}=1.\ln{x}+\frac{x}{\sqrt{x}}.\frac{1}{2\sqrt{x}}\nonumber\\
&y'=\sqrt{x}^{x}\left(\ln{\sqrt{x}}+\frac{1}{2}\right)\nonumber
\end{align}
\end{ex}
