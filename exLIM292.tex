\begin{ex}
\begin{align}
&y=\arccos{\frac{x}{\sqrt{1+x^2}}}\nonumber\\
%resposta
&y'=-\frac{1}{\sqrt{1-\frac{x^2}{1+x^2}}}.\frac{\sqrt{1+x^2}-\frac{x^2}{\sqrt{1+x^2}}}{1+x^2}\nonumber\\
&y'=-\sqrt{1+x^2}.\frac{1+x^2-x^2}{(1-x^2)\sqrt{1+x^2}}=-\frac{1}{1+x^2}\nonumber
\end{align}
\end{ex}
