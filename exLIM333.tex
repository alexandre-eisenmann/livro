\begin{ex}
\begin{align}
&y=x^{e^{x}}\nonumber\\
%resposta
&\ln{y}=\ln{x^{e^{x}}}\quad \Rightarrow \ln{y}=e^{x}.\ln{x}\nonumber\\
&\frac{y'}{y}+e^{x}.\ln{x}+e^{x}.\frac{1}{x}\nonumber\\
&y'=x^{e^{x}}\left[e^{x}\left(\ln{x}+\frac{1}{x}\right)\right]\nonumber
\end{align}
\end{ex}
