\begin{ex}
 (FGV - adaptado) Permutando-se aleatoriamente as letras da palavra ELOGIAR, qual a probabilidade que as letras A e R fiquem juntas em qualquer ordem?
    \begin{enumerate}[(a)]
    \item $\frac{1}{14}$
    \item $\frac{1}{7}$
    \item $\frac{3}{14}$
    \item $\frac{2}{7}$
    \item $\frac{1}{2}$
    \end{enumerate}
       \begin{sol}
       resposta: d \\
       $\frac{6!\cdot2}{7!}=\frac{2}{7}$
       \end{sol}
    
\end{ex}
