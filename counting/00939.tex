\begin{ex}
 	De quantos modos é possível formar a palavra BRASIL nos diagramas abaixo:   
    \begin{enumerate}[(a)]
    \item
    \begin{verbatim}
    B  R  A  S  I  L
    R  A  S  I  L
    A  S  I  L 
    S  I  L 
    I  L
    L
    \end{verbatim}       
    \item 
    \begin{verbatim}
    L  I  S  A  R  B  R  A  S  I  L
       L  I  S  A  R  A  S  I  L
          L  I  S  A  S  I  L
             L  I  S  I  L
                L  I  L
                   L
    \end{verbatim}
    \end{enumerate}  
      \begin{sol}
         \phantom{A}
           \begin{enumerate} [(a)]
               \item $2^5=32$
               \item $3^5=243$
           \end{enumerate}
      \end{sol}
\end{ex}
