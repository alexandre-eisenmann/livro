    \begin{enumerate}[(a)]
    \item $90\cdot0,9^8\cdot0,1^2$
    \item $0,9^8\cdot0,1^2$
    \item $(10!)\cdot0,9^8\cdot0,1^2$
    \item $45\cdot0,9^8\cdot0,1^2$
    \item $9^8\div(10!)$
    \end{enumerate}
      \begin{sol}
        resposta: d \\
      $\mathrm{C}_{{10},8}\cdot(90\%)^8\cdot(10\%)^2=45\cdot(0,9)^8\cdot(0,1)^2$
        
        
      \end{sol}
