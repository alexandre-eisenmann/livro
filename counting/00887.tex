\begin{ex}
 (PucRS) O número de jogos de um campeonato de futebol disputados por n clubes $(n\geq 2)$, no qual todos se enfrentam uma única vez, é:
    \begin{enumerate}[(a)]
    \item $\frac{n^2-n}{2}$
    \item $\frac{n^2}{2}$
    \item $n^2-n$
    \item $n^2$
    \item $n!$
    \end{enumerate}
      \begin{sol}
       resposta: a \\
       $\mathrm{C}_{n,2}=\frac{n!}{2!\cdot(n-2)!}=\frac{n^2-n}{2}$
      \end{sol}
\end{ex}
