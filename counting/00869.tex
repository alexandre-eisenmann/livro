    \begin{enumerate}[(a)]
    \item 70
    \item 84
    \item 140
    \item 210
    \item 252
    \end{enumerate}
      \begin{sol}
       resposta: c \\
       $\mathrm{C}_{2,1}\cdot\mathrm{C}_{8,4}=140$
      \end{sol}
