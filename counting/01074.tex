\begin{ex}
 	(FGV) Quantos números diferentes obtemos reagrupando os algarismos do número 718844?
    \begin{enumerate}[(a)]
    \item 90
    \item 720
    \item 15
    \item 30
    \item 180
    \end{enumerate}
      \begin{sol}
        resposta: e \\
        $\frac{6!}{2!\cdot2!}=180$
      \end{sol}
\end{ex}
