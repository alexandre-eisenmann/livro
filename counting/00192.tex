\begin{ex}
Uma moeda é lançada duas vezes. Qual é a probabilidade de:
   \begin{enumerate}[(a)]
   \item saírem duas caras?
   \item sair apenas uma cara?
   \item não saírem caras?
   \end{enumerate}
    \begin{sol}
     \phantom{A}
      \begin{enumerate} [(a)]
          \item $\frac{1}{2}\cdot\frac{1}{2}=\frac{1}{4}$
          \item $1\cdot\frac{1}{2}=\frac{1}{2}$
          \item $\frac{1}{2}\cdot\frac{1}{2}=\frac{1}{4}$
      \end{enumerate}
    \end{sol}
\end{ex}
