\begin{ex}
 	(Cesgranrio) Se um dado é lançado 3 vezes, a probabilidade de serem obtidos, em qualquer ordem, os valores 1, 2 e 3 é:
    \begin{enumerate}[(a)]
    \item $\frac{1}{108}$
    \item $\frac{1}{72}$
    \item $\frac{1}{216}$
    \item $\frac{1}{120}$
    \item $\frac{1}{36}$
    \end{enumerate}
       \begin{sol}
         resposta: e \\
         $\frac{1}{6}\cdot\frac{1}{6}\cdot\frac{1}{6}\cdot3!=\frac{1}{36}$
       \end{sol}
\end{ex}
