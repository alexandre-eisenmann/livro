\begin{ex}
 (Fuvest) Escolhe-se ao acaso, três vértices distintos de um cubo. A probabilidade de que estes vértices pertençam à mesma face é:
    \begin{enumerate}[(a)]
    \item $\frac{3}{14}$
    \item $\frac{2}{7}$
    \item $\frac{5}{14}$
    \item $\frac{3}{7}$
    \item $\frac{13}{18}$
    \end{enumerate}
      \begin{sol}
        resposta: d \\
        $\frac{6\cdot \mathrm{C}_{4,3}}{\mathrm{C}_{8,3}}=\frac{24}{56}=\frac{3}{7}$
      \end{sol}
\end{ex}
