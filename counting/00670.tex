\begin{ex}
 Quantas soluções inteiras não negativas tem a equação:
x + y + z + t + w = 10?
  \begin{sol}
  \phantom{A} \\
      É um problema de combinação com repetição. \\ Existe um raciocínio alternativo: separamos 10 unidades em 4 partes.
    Total = 10 e 4 sinais de + \\ Usamos a representação abaixo:\\
    
     \begin{tikzpicture}
      \draw[fill=black] (-2,0) circle (0.08);\draw[fill=black] (-1.5,0) circle (0.08);\draw[fill=black] (-1,0) circle (0.08);\draw[fill=black] (-0.5,0) circle (0.08); \draw[fill=black] (0,0) circle (0.08); \draw[fill=black] (0.5,0) circle (0.08); \draw[fill=black] (1,0) circle (0.08);\draw[fill=black] (1.5,0) circle (0.08);\draw[fill=black] (2,0) circle (0.08);\draw[fill=black] (2.5,0) circle (0.08);\draw[very thick] (-1.7,-.5)--(-1.7,.5);\draw[very thick] (-0.2,-.5)--(-0.2,.5);\draw[very thick] (0.8,-.5)--(0.8,.5);\draw[very thick] (2.2,-.5)--(2.2,.5);
     \end{tikzpicture}\\
     
     $\frac{14!}{{10!}.4!}= 1001$
  \end{sol}
\end{ex}
