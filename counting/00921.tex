\begin{ex}
 (Fuvest) Quantos são os números inteiros positivos de cinco algarismos que não tem algarismos adjacentes iguais?
    \begin{enumerate}[(a)]
    \item $5 ^ 9$
    \item $9\cdot 8 ^4$
    \item $8\cdot9^4$
    \item $8^5$
    \item $9^5$
    \end{enumerate}
      \begin{sol}
        resposta: e \\
        $9\cdot9\cdot9\cdot9\cdot9=9^5$
      \end{sol}
\end{ex}
