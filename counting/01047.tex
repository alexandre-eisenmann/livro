    \begin{enumerate}[(a)]
    \item $\frac{1}{30}$
    \item $\frac{1}{24}$
    \item $\frac{1}{20}$
    \item $\frac{7}{120}$
    \item $\frac{7}{720}$
    \end{enumerate}

        $\frac{7}{\mathrm{C}_{{10},3}} = \frac{7}{120}$
        
      \end{sol}
