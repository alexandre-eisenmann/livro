\begin{ex}
 (Puc-RJ) A probabilidade de um dos 100 números (1, 2, 3, 4, ....100) ser múltiplos de 6 e de 10 ao mesmo tempo é:
    \begin{enumerate}[(a)]
    \item 3\%
    \item 6\%
    \item 2\%
    \item 10\%
    \item 60\%
    \end{enumerate} 
      \begin{sol}
        resposta: a \\
        múltiplo de 6 e 10 ao mesmo tempo: 30, 60, 90 $\Longrightarrow p=\frac{3}{100}=3\%$
      \end{sol}
      
\end{ex}
