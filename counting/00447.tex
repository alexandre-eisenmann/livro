\begin{ex}
Lançando-se 4 vezes uma moeda honesta, a probabilidade de que ocorra cara exatamente 3 vezes é:
   \begin{enumerate}[(a)]
   \item $\frac{3}{4}$
   \item $\frac{3}{16}$
   \item $\frac{7}{16}$
   \item $\frac{1}{4}$
   \item $\frac{1}{16}$
   \end{enumerate}
    \begin{sol}
     \phantom{A} \\
     3 caras e 1 coroa $\Longrightarrow \mathrm{C}_{4,3}\cdot (\frac{1}{2})^4=\frac{1}{4}$
    \end{sol}
\end{ex}
