\begin{ex}
(Cesgranrio) Dois eventos, A e B, são mutuamente exclusivos.
Se $P(A\cup B) = 1$ ; $P(A) =3P(B)$, então:
   \begin{enumerate}[(a)]
   \item $P(B) = 0,25$
   \item $P(B) = 0,33$
   \item $P(A) = 0,8$
   \item $P(B) = 0,4$
   \item $P(A) = 0,45$
   \end{enumerate}
     \begin{sol}
      resposta: a \\
      $P(A\cup B)=P(A) + P(B) \rightarrow 1=3P(B)+P(B)\Longrightarrow P(B)=\frac{1}{4}=0,25$
     \end{sol}
\end{ex}
