    \begin{enumerate}[(a)]
    \item 80
    \item 144
    \item 120
    \item 720
    \item 360
    \end{enumerate}
      \begin{sol}
      resposta: c  \\
       \underline{AEO}  \underline{\phantom{AA}}
       \underline{\phantom{AA}} \underline{\phantom{AA}} 
       \underline{\phantom{AA}} $\Longrightarrow 5!=120$
      \end{sol}
\end{ex}
