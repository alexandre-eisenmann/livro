\begin{ex}
 (FGV) Numa sala de reunião há 10 cadeiras e 8 participantes.  De quantas maneiras distintas podem sentar os participantes?
    \begin{enumerate}[(a)]
    \item 181.440
    \item 3.628.800
    \item 1.814.400
    \item 40.320
    \item 403.200
    \end{enumerate}
       \begin{sol}
        resposta: c \\
        $10\cdot9\cdot8\cdot7\cdot6\cdot5\cdot4\cdot3=1.814.400$
       \end{sol}
\end{ex}
