\begin{ex}
 (UFMG) Formam-se comissões de três professores escolhidos entre os sete de uma escola. O número de comissões distintas que podem, assim, ser formadas é:
    \begin{enumerate}[(a)]
    \item 35
    \item 45
    \item 210
    \item $7^2$
    \item 7!
    \end{enumerate}
      \begin{sol}
       resposta: a \\
       $\mathrm{C}_{7,3}=35$
      \end{sol}
\end{ex}
