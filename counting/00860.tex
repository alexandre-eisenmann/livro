\begin{ex}
 Quantos anagramas diferentes podemos formar com as letras das palavras:
    \begin{enumerate}[(a)]
    \item CASA
    \item MATAGAL
    \item BANANA
    \end{enumerate}
      \begin{sol}
       \phantom{A}
         \begin{enumerate} [(a)]
             \item $\frac{4!}{2!}=12$
             \item $\frac{7!}{3!}=840$
             \item $\frac{6!}{3!\cdot2!}=60$
         \end{enumerate}
      \end{sol}
\end{ex}
