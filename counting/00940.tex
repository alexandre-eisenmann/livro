\begin{ex}
 	 Na figura, representamos uma parte do mapa de uma cidade, onde existe um colégio na esquina A e um clube na esquina B. Saindo do colégio e caminhando pelas ruas sempre em direção a B, quantos caminhos existem para chegar ao clube?
 \begin{center}
  \begin{tikzpicture}
   \draw (0,0)rectangle(.8,.5);\draw (1.2,0)rectangle(2,.5);\draw (2.4,0)rectangle(3.2,.5);\draw (3.6,0)rectangle(4.4,.5);\draw (4.8,0)rectangle(5.6,.5); \draw (0,1)rectangle(.8,1.5);\draw (1.2,1)rectangle(2,1.5);\draw (2.4,1)rectangle(3.2,1.5);\draw (3.6,1)rectangle(4.4,1.5);\draw (4.8,1)rectangle(5.6,1.5);
    \draw (0,2)rectangle(.8,2.5);\draw (1.2,2)rectangle(2,2.5);\draw (2.4,2)rectangle(3.2,2.5);\draw (3.6,2)rectangle(4.4,2.5);\draw (4.8,2)rectangle(5.6,2.5); \draw (0,3)rectangle(.8,3.5);\draw (1.2,3)rectangle(2,3.5);\draw (2.4,3)rectangle(3.2,3.5);\draw (3.6,3)rectangle(4.4,3.5);\draw (4.8,3)rectangle(5.6,3.5);
    \draw [fill] (0,0)rectangle(.2,.15);\draw [fill]  (4.8,3)rectangle(5,3.15);\draw node at (0,0) [left] {A};
    \draw node at (4.9,3) [left] {B};
   \end {tikzpicture}
   \end {center}
     \begin{sol}
        \phantom{A}  \\
        $\frac{7!}{3!\cdot4!}=35$
     \end{sol}

\end{ex}
