\begin{ex}
 (Cesgranrio) Em um computador digital “bit” é um dos algarismos 0 ou 1 e uma “palavra” é uma sucessão de “bits”. O número de “palavras” distintas de 32 “bits”, é:
    \begin{enumerate}[(a)]
    \item 2.($2^{32}$-1)
    \item $2^{32}$
    \item $\frac { 32 \text{x} 31  }{2}$
    \item ${32}^2$
    \item 2 x 32
    \end{enumerate}
      \begin{sol}
        resposta: b \\
        32 bits $\Longrightarrow 2^{32}$
      \end{sol}
\end{ex}
