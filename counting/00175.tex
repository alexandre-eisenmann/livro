\begin{ex}
   Um dado será lançado 3 vezes. Qual é a probabilidade de que nas três vezes saia:
   \begin{enumerate}[(a)]
   \item o número 6.
   \item um mesmo número.
   \end{enumerate}
     \begin{sol}
      \phantom{A}
        \begin{enumerate} [(a)]
            \item $\frac{1}{6}\cdot\frac{1}{6}\cdot\frac{1}{6}=\frac{1}{216}$
            \item $1\cdot\frac{1}{6}\cdot\frac{1}{6}=\frac{1}{36}$
        \end{enumerate}
     \end{sol}
\end{ex}
