   \begin{sol}
     \phantom{A}  \\
     de A para B: 7 para a direita e 4 para cima = $\frac{11!}{7!\cdot4!}=330$ \\
     de A para C: 5 para a direita e 3 para cima = $\frac{8!}{5!\cdot3!}=56$  \\
     $p=\frac{56}{330}=\frac{28}{165}$
   \end{sol}
